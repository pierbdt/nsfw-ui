<div class="search">
  <div class="search-icon"><svg>...</svg></div>
  <input class="search-input" x-model="query"
    @input.debounce.200ms="fetchSuggestions()"
    @keydown.escape="showSuggestions = false"
    @keydown.arrow-down.prevent="highlightNext()"
    @keydown.arrow-up.prevent="highlightPrev()"
    @keydown.enter.prevent="selectHighlighted()">
  <div class="search-actions">
    <span class="search-loading" x-show="loading"><svg>...</svg></span>
    <button class="search-clear" x-show="query" @click="query = ''"><svg>...</svg></button>
    <button class="search-submit" x-show="!query" @click="$refs.form.submit()"><svg>...</svg></button>
  </div>

  <div class="search-dropdown" x-show="showSuggestions">
    <template x-for="(item, i) in suggestions">
      <div class="search-item"
        :class="{ 'search-item-active': highlightedIndex === i }">
        <img class="search-item-avatar" :src="item.avatar">
        <div class="search-item-content">
          <span class="search-item-title" x-html="highlight(item.name)"></span>
          <span class="search-item-meta" x-text="item.meta"></span>
        </div>
      </div>
    </template>
    <div class="search-empty" x-show="!suggestions.length">
      No results found.
    </div>
  </div>
</div>

  --color-background: #ffffff;

  --color-muted-foreground: #71717a;
  --color-border: #e4e4e7;

  --color-primary: #18181b;
  --color-primary-foreground: #fafafa;
  --color-secondary: #f4f4f5;
  --color-secondary-foreground: #18181b;
  --color-destructive: #ef4444;
  --color-destructive-foreground: #fafafa;
  --radius-sm: 0.25rem;
  --radius-md: 0.375rem;
  --radius-lg: 0.5rem;
}

Override Example

/* resources/css/app.css */
@import "tailwindcss";
@import "nsfw-ui";

@theme {
  --color-primary: #e11d48;           /* rose */
  --color-primary-foreground: #fff;
  --radius-md: 9999px;                /* pill buttons */
}
